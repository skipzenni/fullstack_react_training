# Lesson 8

## Create User Database

`cd server`

Go to models create Users.js

```bash
module.exports = (sequelize, DataTypes) => {
    const Users = sequelize.define("Users", {
    username: {
        type: DataTypes.STRING,
        allowNull: false,
        },
      password: {
        type: DataTypes.STRING,
        allowNull: false,
      },
    });
    Users.associate = (models) => {
      Users.hasMany(models.Posts, {
        onDelete: 'cascade',
      });
    };
    return Users;
  };
```
